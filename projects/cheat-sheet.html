Ruby Cheat Sheet

Basics

Variables

integer = 3
float = 3.0
string = "three"

Arrays

my_array = [1,2,3]

or

my_array = Array.new(#an integer representing size)

or

my_array = %w{#a list of strings}

Hashes

my_hash = { "key1" => "value1", "key2" => "value2"}

or

my_hash = { symbol1: "value1", symbol2: "value2"}

or

my_hash = Hash.new(#default value)

Methods

def method_name(arguments)
  #do some stuff
end

Objects

class My_Object
  def initialize(variable)
    @variable = variable
  end
end

If-Then

if conditional == true
  #do something
elsif other_conditional == true
  #do something else
else
  #do some third thing
end

Ternary Conditional

test-expression ? if-true-expression : if-false-expression

Case Statements

case some_var
when value1
  #do something
when value2
  #do something else
else
  #do some third thing
end

Loops

While

while conditional == true do
  #do something
end

Until

until conditional == false do
  #do something
end

For

for variable in collection do
  #do something
end

Some Collection Methods

.include?

my_array.include?("element")

Returns true if a hash or array includes the element.

.find

my_array.find {|n| n == true}

Returns the first element in the array that returns true on the code block

.select, .find_all

my_array.select {|n| n == true}

Returns an array of all values that match the code block. The reverse is .reject.

.group_by

my_array.group_by {|n| n.length}

Returns a hash. The result will become the key, the value will be an array of each value that matches each value to the resulting key.

.partition

my_array.partition {|element| element == true}

Returns two arrays: a false array and a true array.

.sample

my_array.sample(#some number of objects)

Returns a random sample of n unique objects from the collection.

.take, .drop

my_array.take(2)
my_array.drop(2)

Take returns an array of that many. Drop returns collection without that many.

.values_at

my_hash.values_at("key1","key2")

Returns an array at the values at the keys provided as an argument.

Each and Relatives

.each

my_array.each do |element| element.upcase end

Calls the block of code on each element in the collection.

.reverse_each

my_array.reverse_each {|n| puts n * 10}

Iterates backward through an enumerable collection.

.each_with_index

my_array.each_with_index {|element,index| puts "#{index}:#{element}"}

Provides both the item and its index each time through the code block.

.cycle

my_array.cycle(#some number of times) do |element|

Cycles the object through the code block the number of times you specify.

.map, .map!

my_array.map {|element| element.upcase}

Always returns an array that is always the same size as the original colllection after having called the code block on each item.
